def Lorenz(x, A, σ, μ):
    return A / ((x - μ)**2/σ + 1)

def Gauss(x, A, σ, μ):
    return A*np.exp(-(x-μ)**2/(2*σ)**2)

def generate_function_data(x, param_extent, N):
    '''
    Create data set for Gaussean and Lorentzian functions
    inputs  : 
    x            : 1D array of x-axis value for which we need the functional values
    param_extent : [A_min, A_max, σ_min, σ_max, μ_min, μ_max]
    N            : Number of functions needed
    
    outputs :
    Y            : list of functional values corresponding to x
    labels       : type of function for each y
    '''
    labels = np.random.randint(low=0, high=2, size=N)
    A      = np.random.uniform(low=param_extent[0], high=param_extent[1], size=N)
    σ      = np.random.uniform(low=param_extent[2], high=param_extent[3], size=N)
    μ      = np.random.uniform(low=param_extent[4], high=param_extent[5], size=N)

    # print(μ)

    Y =  [None]*N

    for i in range(N):
        Y[i] = (1 - labels[i])*Lorenz(x, A[i], σ[i], μ[i]) + labels[i]*Gauss(x, A[i], σ[i], μ[i])
    
    return np.array(Y),labels

from sklearn.model_selection import train_test_split

N = 10 # Number of functions to generate
extent = [0., 1., 0., 0.5, -2., 2.] #A, σ, μ : Amplitude, spread, center

x = np.arange(-5,5,0.1)
func, labels = generate_function_data(x, extent, N)

x_train, x_test, y_train, y_test = train_test_split(func, labels, test_size=0.2)
